# Event 1000001: Accepted minority of noculture
# Events 1000002-1000409: Country speaks noculture (fixed through capital's majority culture)

province_event = {
	id = 1000000
	title = EVTNAME000000
	desc = EVTDESC000000
	picture = "Demonstration"
	
	trigger = {
		any_pop = {
			OR = {
				has_pop_culture = noculture
				has_pop_culture = dummy
			}
		}
	}
	
	mean_time_to_happen = {
		months = 0.001
	}
	
	option = {
		name = EVTOPTA000000
		any_pop = {
			limit = { OR = { has_pop_culture = noculture has_pop_culture = dummy } }
			reduce_pop = -1
		}
	}
}
