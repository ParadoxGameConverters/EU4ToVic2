country_event = {
	id = 601999
	title = EVTNAMEGER
	desc = EVTDESCGER
	mean_time_to_happen = {
		months = 2
	}
	trigger = {
		OR = {
			AND = {
				primary_culture = north_german
				NOT = { accepted_culture = south_german }
			}
			AND = {
				primary_culture = south_german
				NOT = { accepted_culture = north_german }
			}
		}
	}
	option = {
		name = EVTOPTAGER
	}
	immediate = {
		add_accepted_culture = north_german
		add_accepted_culture = south_german
	}
}

### National countries events
#These events are useful for CK2 conversions
#because they make it possibly for a country of X culture
#to become the X country

country_event = {
	id = 601001
	title = EVTNAME601xxx
	desc = EVTDESC601xxx
	major = yes
	allow_multiple_instances = no
	
	trigger = {
		civilized = yes
		is_independant = yes
		war = no
		number_of_states = 2
		has_recently_lost_war = no
		nationalism_n_imperialism = 1
		primary_culture = flemish
		NOT = { exists = BEL } #Cultural union
		NOT = { exists = NET } #Cultural union
		NOT = { exists = FLA } #Cultural country
		FLA = {
			all_core = {
				OR = {
					owned_by = THIS
					owner = {
						NOT = { primary_culture = flemish }
						accepted_culture = flemish
						in_sphere = THIS
					}
					owner = {
						NOT = { primary_culture = flemish }
						NOT = { accepted_culture = flemish }
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = {
		name = STRING_OK
		change_tag = FLA
		prestige = 5
		any_country = {
			limit = {
				in_sphere = FROM
				accepted_culture = flemish
			}
			inherit = FROM
		}
	}
	
}
